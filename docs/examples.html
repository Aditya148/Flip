Practical code examples from basic to advanced usage

1. Basic Usage

The simplest way to use Flip:

from flip import Flip

# Initialize with a directory
flip = Flip(directory="./sample_docs")

# Query your documents
response = flip.query("What is the main topic?")

print(response.answer)
print(f"Sources: {len(response.citations)} documents")

2. Advanced Configuration

Custom configuration with multiple providers:

from flip import Flip, FlipConfig

# Custom configuration
config = FlipConfig(
    llm_provider="openai",
    llm_model="gpt-4-turbo-preview",
    embedding_provider="openai",
    chunking_strategy="semantic",
    chunk_size=512,
    retrieval_top_k=5,
    use_hybrid_search=True,
    use_reranking=True,
    enable_cache=True
)

flip = Flip(directory="./docs", config=config)
response = flip.query("What are the key features?")

3. Azure OpenAI

Using Azure OpenAI for both LLM and embeddings:

from flip import Flip, FlipConfig

config = FlipConfig(
    llm_provider="azure-openai",
    llm_model="gpt-4",  # Your deployment name
    
    embedding_provider="azure-openai",
    embedding_model="text-embedding-ada-002",  # Your deployment name
    
    azure_openai_api_key="your-key",
    azure_openai_endpoint="https://your-resource.openai.azure.com/",
    
    use_hybrid_search=True,
    use_reranking=True
)

flip = Flip(directory="./docs", config=config)
response = flip.query("What is AI?")

4. Multiple Providers

Try different providers and compare results:

from flip import Flip

providers = ["openai", "anthropic", "google"]

for provider in providers:
    flip = Flip(directory="./docs", llm_provider=provider)
    response = flip.query("Summarize the main points")
    print(f"{provider}: {response.answer[:100]}...")

5. Using EmbeddingGemma

Use Google's EmbeddingGemma model:

from flip import Flip, FlipConfig

config = FlipConfig(
    embedding_provider="huggingface",
    embedding_model="google/embeddinggemma-300m",
    llm_provider="openai",  # or any other provider
    use_hybrid_search=True,
    use_reranking=True
)

flip = Flip(directory="./docs", config=config)
response = flip.query("What are the key points?")

6. Incremental Updates

Efficiently update your index with changed documents:

from flip import Flip

# Initial indexing
flip = Flip(directory="./docs")

# Later, refresh with only changed documents
stats = flip.refresh_index()

print(f"Added: {stats['added']}")
print(f"Updated: {stats['updated']}")
print(f"Deleted: {stats['deleted']}")

7. Evaluation

Evaluate RAG performance with metrics:

from flip import Flip

flip = Flip(directory="./docs")

# Evaluate a query
result = flip.evaluate(
    query="What is AI?",
    relevant_doc_ids=["chunk_1", "chunk_2"],  # Ground truth
    k=5
)

print(f"Precision: {result.retrieval_precision:.3f}")
print(f"Recall: {result.retrieval_recall:.3f}")
print(f"F1 Score: {result.retrieval_f1:.3f}")
print(f"Overall Score: {result.overall_score:.3f}")

8. Logging and Debugging

Debug and profile your RAG pipeline:

from flip import Flip
from flip.utils.logger import setup_logging
from flip.utils.debug import FlipDebugger, profile_query

# Setup logging
logger = setup_logging(level="DEBUG", log_file="./logs/flip.log")

# Setup debugging
debugger = FlipDebugger(enabled=True)

# Initialize Flip
flip = Flip(directory="./docs")

# Profile a query
profile_query(flip, "What is AI?", iterations=10)

# Check timings
debugger.print_timings()

9. Local Models (No API Keys)

Use Ollama and Sentence Transformers for completely local RAG:

from flip import Flip, FlipConfig

config = FlipConfig(
    llm_provider="ollama",
    llm_model="llama2",
    
    embedding_provider="sentence-transformers",
    embedding_model="all-MiniLM-L6-v2",
    
    use_hybrid_search=True
)

flip = Flip(directory="./docs", config=config)
response = flip.query("What is this about?")
